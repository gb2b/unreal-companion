def build_multimodal_message(
    text: str,
    attachments: List[Attachment]
) -> Message:
    """Construit un message avec attachments."""

    content = []

    # Texte principal
    content.append({"type": "text", "text": text})

    # Images
    for att in attachments:
        if att.type == "image":
            content.append({
                "type": "image",
                "source": {"type": "base64", "data": att.base64}
            })

    return Message(content=content)
